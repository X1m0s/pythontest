# ═══════════════════════════════════════════════════════
# 一、无参可用，传参无特殊限制
#   任意类型均可传入，构造函数会尝试转换
# ═══════════════════════════════════════════════════════

int(3.14)           # 3      浮点 → 截断小数部分，不四舍五入
int("100")          # 100    字符串 → 整数字面量必须合法
int()               # 0      无参默认返回 0

float(3)            # 3.0    整数自动补小数位
float("3.14")       # 3.14   字符串须可解析为数字
float()             # 0.0    无参默认返回 0.0

complex(1, 2)       # (1+2j) 分别传实部和虚部
complex("1+2j")     # (1+2j) 字符串解析（注意+ - 旁不能有空格）
complex()           # 0j     无参默认返回 0j

str(123)            # '123'  调用对象的 __str__() 方法
str([1, 2])         # '[1, 2]'  非字符串类型统一转为字符串表示
str()               # ''     无参返回空字符串

bool(1)             # True   非零数值 → True
bool(0)             # False  零 → False
bool("")            # False  空字符串 → False
bool("a")           # True   非空字符串 → True
bool([])            # False  空容器 → False
bool([1])           # True   非空容器 → True
bool()              # False  无参默认 False


# ═══════════════════════════════════════════════════════
# 二、无参 → 空容器；传参 → 参数必须是可迭代对象
#   只要传了参数就必须可迭代，int/float/bool/None 不行
# ═══════════════════════════════════════════════════════

list([1, 2, 3])     # [1, 2, 3]      元组、集合、字符串等均可
list("abc")         # ['a', 'b', 'c'] 字符串逐字符拆分
list()              # []             无参创建空列表
# list(42)          # TypeError      int 不可迭代

tuple([1, 2, 3])    # (1, 2, 3)      和 list() 规则完全一致
tuple("abc")        # ('a', 'b', 'c')
tuple()             # ()
# tuple(3.14)       # TypeError      同样要求可迭代

set([1, 2, 2, 3])   # {1, 2, 3}      自动去重；注意结果无序
set("hello")        # {'h', 'e', 'l', 'o'}  重复的 'l' 只保留一个
set()               # set()          空集合（注意不是 {}，{} 是空字典）
# set(True)         # TypeError      bool 不可迭代

frozenset([1, 2])   # frozenset({1, 2})  不可变集合，规则同 set
frozenset()

bytes([65, 66, 67]) # b'ABC'        整数列表 → 字节，每个值须在 0~255
bytes()             # b''           空字节
# bytes(42)         # TypeError

bytearray([65, 66]) # bytearray(b'AB')  可变版 bytes，规则相同
bytearray()


# ═══════════════════════════════════════════════════════
# 三、无参 → 空字典；传参 → 三点要求
#   (1) 参数可迭代
#   (2) 每个子元素恰好两个元素
#   (3) 键（第一个元素）必须可哈希
# ═══════════════════════════════════════════════════════

dict([("a", 1), ("b", 2)])      # {'a': 1, 'b': 2}
dict([["a", 1], ["b", 2]])      # {'a': 1, 'b': 2}  列表也行
dict(a=1, b=2)                  # {'a': 1, 'b': 2}  关键字参数方式（不要求可迭代）
dict()                          # {}               无参创建空字典
# dict([("a", 1, "多余")])       # ValueError       子元素必须恰好 2 个
# dict(42)                       # TypeError        参数不可迭代


# ═══════════════════════════════════════════════════════
# 四、set / frozenset：额外要求元素可哈希
#   可哈希的：int、float、str、tuple（内嵌也须全可哈希）
#   不可哈希的：list、set、dict、bytearray
# ═══════════════════════════════════════════════════════

set([1, 2, 3])              # ✓ 整数可哈希
set(["a", "b"])             # ✓ 字符串可哈希
set([(1, 2), (3, 4)])       # ✓ 元组可哈希（内部也是整数）
# set([1, [2], 3])          # TypeError 列表不可哈希
# set([1, {"a": 1}])        # TypeError 字典不可哈希


# ═══════════════════════════════════════════════════════
# 五、range：无参报错，参数必须全部是整数
# ═══════════════════════════════════════════════════════

range(5)            # range(0, 5)     一个参数 → stop（从 0 开始，不含 5）
range(2, 5)         # range(2, 5)     两个参数 → start, stop
range(0, 10, 2)     # range(0, 10, 2) 三个参数 → start, stop, step
# range()           # TypeError      必须至少 1 个参数
# range(1.5)        # TypeError      浮点不行，必须 int
# range("5")        # TypeError      字符串不行


# ═══════════════════════════════════════════════════════
# 六、int(string, base)：base 范围限制 2~36
#   不传 base 时默认 base=10
# ═══════════════════════════════════════════════════════

int("1010", 2)      # 10     二进制字符串 → 十进制
int("ff", 16)       # 255    十六进制
int("77", 8)        # 63     八进制
int("10", 3)        # 3      三进制
# int("10", 37)     # ValueError  base 必须在 2~36 之间
# int("10", 1)      # ValueError
# int("", 2)        # ValueError  空字符串无法解析


# ═══════════════════════════════════════════════════════
# 七、bytes / bytearray：字符串转字节必须指定编码
#   整数列表方式不需要编码参数
# ═══════════════════════════════════════════════════════

bytes("hi", "utf-8")        # b'hi'    字符串 + 编码
bytes("你好", "utf-8")      # b'\xe4\xbd\xa0\xe5\xa5\xbd'
bytes([104, 105])           # b'hi'    整数列表（0~255），不需要编码
bytes()                     # b''
# bytes("hi")               # TypeError 字符串必须传 encoding


# ═══════════════════════════════════════════════════════
# 八、float(string)：字符串必须能解析为浮点数
#   几个特殊值例外
# ═══════════════════════════════════════════════════════

float("3.14")           # 3.14
float("1e-3")           # 0.001    科学记数法
float("inf")            # inf      正无穷（特殊值）
float("-inf")           # -inf     负无穷
float("nan")            # nan      非数值
# float("abc")          # ValueError
# float("12.34.56")     # ValueError


# ═══════════════════════════════════════════════════════
# 九、complex(string)：+ - 号两边绝对不能有空格
#   分别传实部虚部时无此限制
# ═══════════════════════════════════════════════════════

complex(1, 2)           # (1+2j)   分两步传，无空格问题
complex("1+2j")         # (1+2j)   没有空格 → ✓
complex("1-2j")         # (1-2j)   ✓
# complex("1 + 2j")     # ValueError  有空格 → ✗
# complex("1 +2j")      # ValueError  半边空格也不行
# complex("1+ 2j")      # ValueError


# ═══════════════════════════════════════════════════════
# 十、bool：注意 True/False 与 1/0 的关系
#   在数值场景下 True==1、False==0，集合会去重
# ═══════════════════════════════════════════════════════

bool("任意非空")       # True
bool("")               # False
bool(0)                # False
bool(1)                # True
# 以下为常见坑点
True == 1              # True   — 比较相等
True is 1              # False  — 但不是一个对象
{1, True}              # {1}    — 集合去重（哈希相同）
{True, 1}              # {True} — 后放入的被去重
