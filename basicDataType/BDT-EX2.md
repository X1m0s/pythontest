# ═══════════════════════════════════════════════════════
# int(x, base=10)
#   将 x 转为整数；base 指定进制（2~36），仅 x 为字符串时有效
# ═══════════════════════════════════════════════════════
print(int(3.14))            # 3         浮点截断，不四舍五入
print(int("100"))           # 100       十进制字符串
print(int("1010", 2))       # 10        二进制字符串 → 十进制
print(int("ff", 16))        # 255       十六进制
print(int("0xf", 0))        # 15        base=0 自动从前缀推断
# print(int("abc"))         # ValueError
# print(int("1010", 37))    # ValueError  base 须 2~36

# ═══════════════════════════════════════════════════════
# float(x)
#   将 x 转为浮点数；字符串须可解析为数字
# ═══════════════════════════════════════════════════════
print(float(3))             # 3.0       整数自动补小数
print(float("3.14"))        # 3.14      常规字符串
print(float("1e-3"))        # 0.001     科学记数法
print(float("inf"))         # inf       正无穷（特殊值）
print(float("-inf"))        # -inf      负无穷
print(float("nan"))         # nan       非数值
# print(float("abc"))       # ValueError

# ═══════════════════════════════════════════════════════
# complex(real, imag) / complex("real+imagj")
#   创建复数；字符串中加号两边不能有空格
# ═══════════════════════════════════════════════════════
print(complex(1, 2))        # (1+2j)    分传实部虚部
print(complex("1+2j"))      # (1+2j)    字符串解析
print(complex("1-2j"))      # (1-2j)
print(complex("3j"))        # 3j        纯虚数
print(complex(5))           # (5+0j)    只传实部
# print(complex("1 + 2j"))  # ValueError  空格导致解析失败

# ═══════════════════════════════════════════════════════
# str(x)
#   将任意对象转为字符串；调用对象的 __str__()
# ═══════════════════════════════════════════════════════
print(str(123))             # 123
print(str(3.14))            # 3.14
print(str([1, 2, 3]))       # [1, 2, 3]
print(str(None))            # None
print(str(True))            # True

# ═══════════════════════════════════════════════════════
# repr(x)
#   将对象转为"表达式字符串"；调用 __repr__()，适合调试
# ═══════════════════════════════════════════════════════
print(repr("hello"))        # 'hello'        多一层引号
print(repr(123))            # 123            数字原样
print(repr([1, 2]))         # [1, 2]
print(repr(None))           # None
# str vs repr 对字符串的区别
print(str("abc"))           # abc            →
print(repr("abc"))          # 'abc'          ← 多出引号，能直接放进代码

# ═══════════════════════════════════════════════════════
# eval(expression, globals, locals)
#   执行字符串中的 Python 表达式，返回结果
#   注意：永不对外部输入使用 eval()，存在安全风险
# ═══════════════════════════════════════════════════════
print(eval("1 + 2"))        # 3
print(eval("[1, 2, 3]"))    # [1, 2, 3]
print(eval("len('abc')"))   # 3
x = 10
print(eval("x * 2"))        # 20        引用外部变量
# eval("__import__('os').system('dir')")  # 危险！绝不用 eval 执行不可信输入

# ═══════════════════════════════════════════════════════
# tuple(iterable)
#   将可迭代对象转为元组（不可变）
# ═══════════════════════════════════════════════════════
print(tuple([1, 2, 3]))     # (1, 2, 3)  列表 → 元组
print(tuple("abc"))         # ('a', 'b', 'c')  字符串逐字
print(tuple({10, 20}))      # (10, 20)   集合 → 元组（无序）
print(tuple({"a": 1}))      # ('a',)     字典 → 取键
print(tuple())              # ()         空元组

# ═══════════════════════════════════════════════════════
# list(iterable)
#   将可迭代对象转为列表（可变）
# ═══════════════════════════════════════════════════════
print(list((1, 2, 3)))      # [1, 2, 3]  元组 → 列表
print(list("abc"))          # ['a', 'b', 'c']  字符串逐字
print(list({"a": 1}))       # ['a']      字典 → 取键
print(list(range(3)))       # [0, 1, 2]  range → 列表
print(list())               # []         空列表

# ═══════════════════════════════════════════════════════
# set(iterable)
#   将可迭代对象转为集合；去重，无序，元素须可哈希
# ═══════════════════════════════════════════════════════
print(set([1, 2, 2, 3]))    # {1, 2, 3}         列表去重
print(set("hello"))         # {'h', 'e', 'l', 'o'}  字符串去重
print(set((1, 2, 3)))       # {1, 2, 3}         元组
print(set())                # set()             空集合（不是 {}）
# print(set([1, [2]]))      # TypeError        list 不可哈希

# ═══════════════════════════════════════════════════════
# dict(mapping, **kwargs) / dict(iterable, **kwargs)
#   创建字典；可迭代中每个元素须恰好两个子元素；键须可哈希
# ═══════════════════════════════════════════════════════
print(dict([("a", 1), ("b", 2)]))   # {'a': 1, 'b': 2}    元组列表
print(dict([["c", 3], ["d", 4]]))  # {'c': 3, 'd': 4}    列表的列表也行
print(dict(a=1, b=2))               # {'a': 1, 'b': 2}    关键字参数
print(dict())                       # {}                  空字典
print(dict({"x": 1}))               # {'x': 1}            从已有字典拷贝
# print(dict([("a", 1, 2)]))        # ValueError          子元素须恰好2个

# ═══════════════════════════════════════════════════════
# frozenset(iterable)
#   转为不可变集合；创建后不能再增删元素；其他规则同 set
# ═══════════════════════════════════════════════════════
print(frozenset([1, 2, 2, 3]))      # frozenset({1, 2, 3})
print(frozenset("abc"))             # frozenset({'a', 'b', 'c'})
print(frozenset())                  # frozenset()
# 不可变 → 可作为 dict 的键
d = {frozenset([1, 2]): "ok"}
print(d[ frozenset([1, 2]) ])       # ok
# 对比：
# d = {set([1, 2]): "fail"}         # TypeError  set 可哈希

# ═══════════════════════════════════════════════════════
# chr(i)
#   将 Unicode 码点（整数）转为对应的单字符
#   i 范围：0 ~ 1,114,111（0x10FFFF）
# ═══════════════════════════════════════════════════════
print(chr(65))              # A            大写 A
print(chr(97))              # a            小写 a
print(chr(48))              # 0            数字 0
print(chr(20013))           # 中           中文
print(chr(0x4E2D))          # 中           十六进制写法
# print(chr(-1))            # ValueError   不允许负数
# print(chr(1114112))       # ValueError   超出码点范围

# ═══════════════════════════════════════════════════════
# ord(c)
#   将单字符转为其 Unicode 码点（整数）
#   c 必须是恰好一个字符的字符串
# ═══════════════════════════════════════════════════════
print(ord("A"))             # 65
print(ord("a"))             # 97
print(ord("0"))             # 48
print(ord("中"))            # 20013
print(hex(ord("中")))       # 0x4e2d       配合 hex 看十六进制码点
# print(ord("ab"))          # TypeError    必须是单字符，不能多
# print(ord(""))            # TypeError    空字符串也不行

# chr 和 ord 互逆
print(chr(ord("A")))        # A
print(ord(chr(65)))         # 65

# ═══════════════════════════════════════════════════════
# hex(x)
#   将整数转为十六进制字符串，带 "0x" 前缀
#   x 必须是整数
# ═══════════════════════════════════════════════════════
print(hex(255))             # 0xff
print(hex(16))              # 0x10
print(hex(0))               # 0x0
print(hex(-42))             # -0x2a       负数 → 字符串前加 -
# print(hex("ff"))          # TypeError   必须是 int，不能是字符串
# print(hex(3.14))          # TypeError   浮点不行

# ═══════════════════════════════════════════════════════
# oct(x)
#   将整数转为八进制字符串，带 "0o" 前缀
#   x 必须是整数
# ═══════════════════════════════════════════════════════
print(oct(8))               # 0o10
print(oct(64))              # 0o100
print(oct(0))               # 0o0
print(oct(-10))             # -0o12       负数
# print(oct("64"))          # TypeError   必须是 int
# print(oct(3.14))          # TypeError   浮点不行

# 三者配合：十进制 ↔ 其他进制
n = 42
print(hex(n))               # 0x2a        十进制 → 十六进制
print(oct(n))               # 0o52        十进制 → 八进制
print(int("0x2a", 16))      # 42          十六进制字符串 → 十进制
print(int("0o52", 8))       # 42          八进制字符串 → 十进制
